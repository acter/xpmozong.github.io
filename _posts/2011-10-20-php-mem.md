<p>先来一个test1.php</p>   
    class ABC{
        public $numBox = array();
        public function ddd(){
            for($i = 1; $i <= 600; $i++){
                $this->numBox[] = mt_rand(1,1000000);
            }
            echo 'Now memory_get_usage: '.memory_get_usage().'<br />';
        }
    }
    $instance = new ABC();
    for($i = 0; $i< 50; $i++){
        $instance->ddd();
    }
